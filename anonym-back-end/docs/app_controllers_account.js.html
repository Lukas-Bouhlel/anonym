/**
 * Rﾃｩcupﾃｨre les informations du compte de l'utilisateur connectﾃｩ, y compris l'inventaire actif.
 *
 * @function readAccount
 * @async
 * @param {Object} req - La requﾃｪte HTTP.
 * @param {Object} res - La rﾃｩponse HTTP.
 * @throws {Error} En cas d'erreur lors de la rﾃｩcupﾃｩration des informations de l'utilisateur.
 */
exports.readAccount = async (req, res) => {
    try {
        const userId = req.auth.userId;// Rﾃｩcupﾃｩrer l'ID de l'utilisateur depuis les paramﾃｨtres JWT

        // Trouver l'utilisateur par ID avec findOne
        const user = await User.findOne({
            where: { id: userId },
            include: [
                {
                    model: Inventory,
                    where: { active: true }, // Ne rﾃｩcupﾃｩrer que les articles actifs
                    include: [
                        {
                            model: Shop, // Inclure les informations sur l'article depuis le modﾃｨle Shop
                            attributes: ['title', 'type', 'content', 'amount']
                        }
                    ],
                    required: false
                }
            ]
        });

        if (!user) {
            return res.status(404).json({ message: "User not found" });
        }

        res.status(200).json(user);
    } catch (error) {
        res.status(500).json({
            message: error.message || 'Une erreur est survenue lors de la rﾃｩcupﾃｩration des comptes.'
        });
    }
};

/**
 * Met ﾃ jour les informations d'un utilisateur.
 *
 * @function update
 * @async
 * @param {Object} req - La requﾃｪte HTTP.
 * @param {Object} res - La rﾃｩponse HTTP.
 * @throws {Error} En cas d'erreur lors de la mise ﾃ jour de l'utilisateur.
 *
 * @example
 * // Exemple de requﾃｪte
 * PATCH /api/users/update
 * {
 *   "datas": {
 *     "username": "nouveau_nom",
 *     "email": "nouvel_email@example.com",
 *     "avatar": "delete"
 *   }
 * }
 */
exports.update = async (req, res) => {
    try {
        const userId = req.auth.userId;// Rﾃｩcupﾃｩrer l'ID de l'utilisateur depuis les paramﾃｨtres JWT
        const datas = JSON.parse(req.body.datas);
        const { username, email, avatar } = datas;

        if (!userId) {
            return res.status(400).json({ message: "User ID is required." });
        }

        // Trouver l'utilisateur par ID
        const user = await User.findOne({ where: { id: userId } });

        if (!user) {
            return res.status(404).json({ message: "User not found" });
        }

        let newAvatarPath = user.avatar;
        // Vﾃｩrifier si l'avatar doit ﾃｪtre supprimﾃｩ
        if (req.avatarData && avatar === "delete") {
            // Supprimer l'ancien avatar
            if (user.avatar) {
                const oldAvatarPath = path.join(__dirname, '../../uploads/profiles/avatars', path.basename(user.avatar));
                if (fs.existsSync(oldAvatarPath)) {
                    fs.unlinkSync(oldAvatarPath); // Supprimer l'ancien fichier
                }
            }

            // Gﾃｩnﾃｩrer un nouvel avatar par dﾃｩfaut avec couleur modifiﾃｩe
            if (!req.file && req.avatarData) {
                const { circleColor, pathColor, uniqueAvatarName } = req.avatarData;

                // Chemin du fichier avatar par dﾃｩfaut
                const defaultAvatarPath = path.join(__dirname, '../../uploads/profiles/default/default_avatar.svg');
                const userAvatarPath = path.resolve(__dirname, '../../uploads/profiles/avatars', uniqueAvatarName);

                // Lire le contenu du SVG
                let svgContent = fs.readFileSync(defaultAvatarPath, 'utf8');

                // Remplacer la couleur dans le SVG
                svgContent = svgContent.replace(/<circle[^>]*fill="[^"]*"[^>]*>/, `<circle cx="115" cy="115" r="115" fill="${circleColor}"/>`);
                svgContent = svgContent.replace(/<path[^>]*fill="[^"]*"[^>]*>/, `<path d="M114.37 48L150.593 117.743L167.732 116.319L184.87 114.894L158.396 132.766C169.932 154.979 184.87 183.741 184.87 183.741L161.077 183.801L140.549 144.814L66.4727 183.801H44L54.9652 162.64L135.365 133.027C135.365 133.027 135.396 133.016 135.457 132.994C136.576 132.584 177.708 117.529 184.87 114.894L167.732 116.319L150.593 117.743L131.372 124.824L115.018 92.3567L103.054 114.894L89.6156 140.207L44 157.012L66.0193 141.308L79.1852 115.9L114.37 48Z" fill="${pathColor}"/>`);

                // Enregistrer le SVG modifiﾃｩ
                fs.writeFileSync(userAvatarPath, svgContent);
  
                // Mettre ﾃ jour l'avatar de l'utilisateur dans la base de donnﾃｩes
                newAvatarPath = `${req.protocol}://${req.get("host")}/uploads/profiles/avatars/${uniqueAvatarName}`;
            }
        } else if (req.file) {
            // Dﾃｩfinir le chemin du nouvel avatar
            newAvatarPath = `${req.protocol}://${req.get("host")}/uploads/profiles/avatars/${req.file.filename}`;

            // Supprimer l'ancien avatar
            if (user.avatar) {
                const oldAvatarPath = path.join(__dirname, '../../uploads/profiles/avatars', path.basename(user.avatar));
                if (fs.existsSync(oldAvatarPath)) {
                    fs.unlinkSync(oldAvatarPath); // Supprimer le fichier
                }
            }
        }

        // Mise ﾃ jour des informations de l'utilisateur
        if (username) {
            // Valider le username
            const usernameRegex = /^(?=.*[a-zA-Z])(?=^[A-Za-z0-9_-]{3,15}$)/; // Regex pour le username
            if (!usernameRegex.test(username)) {
                return res.status(400).json({
                    message: "Nom d'utilisateur : 3 ﾃ 15 caractﾃｨres, min une lettre, et peut contenir des chiffres, des tirets et des tirets du bas"
                });
            }
            user.username = username;
        }
        if (email) user.email = email;
        user.avatar = newAvatarPath;

        await user.save();

        res.status(200).json(user);
    } catch (error) {
        res.status(500).json({
            message: error.message || 'Une erreur est survenue lors de la mise ﾃ jour du compte.'
        });
    }
};

/**
 * Met ﾃ jour le mot de passe d'un utilisateur.
 *
 * @function updatePassword
 * @async
 * @param {Object} req - La requﾃｪte HTTP.
 * @param {Object} res - La rﾃｩponse HTTP.
 * @throws {Error} En cas d'erreur lors de la mise ﾃ jour du mot de passe.
 */
exports.updatePassword = async (req, res) => {
    try {
        const userId = req.auth.userId; // Rﾃｩcupﾃｩrer l'ID de l'utilisateur via JWT
        const { currentPassword, newPassword, confirmNewPassword } = req.body;

        // Vﾃｩrification des champs
        if (!currentPassword || !newPassword || !confirmNewPassword) {
            return res.status(400).json({ message: "Tous les champs sont obligatoires." });
        }

        // Vﾃｩrifier que le nouveau mot de passe correspond ﾃ la confirmation
        if (newPassword !== confirmNewPassword) {
            return res.status(400).json({ message: "Les nouveaux mots de passe ne correspondent pas." });
        }

        // Regex pour valider le mot de passe
        const passwordRegex = /^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)(?=.*[!@#$%^&*()_+=\-[\]{};:,.<>?/\\|`~"'ﾂ｣ﾂ､ﾂｧﾂｵﾂ｢竄ｹ])[A-Za-z\d!@#$%^&*()_+=\-[\]{};:,.<>?/\\|`~"'ﾂ｣ﾂ､ﾂｧﾂｵﾂ｢竄ｹ]{12,}$/;
        if (!passwordRegex.test(newPassword)) {
            return res.status(400).json({ message: "Mot de passe : 12 caractﾃｨres min, avec majuscules, minuscules, chiffres et caractﾃｨres spﾃｩciaux" });
        }

        // Trouver l'utilisateur dans la base de donnﾃｩes
        const user = await User.findOne({ where: { id: userId } });

        if (!user) {
            return res.status(404).json({ message: "Utilisateur non trouvﾃｩ." });
        }

        // Vﾃｩrifier que le mot de passe actuel est correct
        const passwordMatch = await bcrypt.compare(currentPassword, user.password);
        if (!passwordMatch) {
            return res.status(401).json({ message: "Mot de passe actuel incorrect." });
        }

        // Hacher le nouveau mot de passe
        const hashedNewPassword = await bcrypt.hash(newPassword, 10);

        // Mettre ﾃ jour le mot de passe de l'utilisateur
        user.password = hashedNewPassword;
        await user.save();

        res.status(200).json({ message: "Mot de passe mis ﾃ jour avec succﾃｨs." });
    } catch (error) {
        res.status(500).json({
            message: error.message || "Une erreur est survenue lors de la mise ﾃ jour du mot de passe."
        });
    }
};

/**
 * Supprime un utilisateur et son avatar.
 *
 * @function delete
 * @async
 * @param {Object} req - La requﾃｪte HTTP.
 * @param {Object} res - La rﾃｩponse HTTP.
 * @throws {Error} En cas d'erreur lors de la suppression de l'utilisateur.
 */
exports.delete = async (req, res) => {
    try {
        const userId = req.auth.userId;// Rﾃｩcupﾃｩrer l'ID de l'utilisateur depuis les paramﾃｨtres JWT

        if (!userId) {
            return res.status(400).json({ message: "User ID is required." });
        }

        // Trouver l'utilisateur par ID
        const user = await User.findOne({ where: { id: userId } });

        if (!user) {
            return res.status(404).json({ message: "User not found" });
        }

        // Supprimer l'avatar
        if (user.avatar) {
            const avatarPath = path.join(__dirname, '../../uploads/profiles/avatars', path.basename(user.avatar));
            fs.unlink(avatarPath, (err) => {
                if (err) {
                    console.error("Error deleting avatar file:", err);
                }
            });
        }

        res.clearCookie('token', {
            httpOnly: true,
            secure: process.env.NODE_ENV === 'production',
            sameSite: 'Strict',
        });

        // Supprimer l'utilisateur
        await user.destroy();

        res.status(200).json({ message: "User deleted successfully" });
    } catch (error) {
        res.status(500).json({
            message: error.message || 'Une erreur est survenue lors de la suppression du compte.'
        });
    }
};
